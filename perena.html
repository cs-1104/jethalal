from sklearn.compose import ColumnTransformer

new_columns = ct.get_feature_names_out()

encoded_df = pd.DataFrame(encoded_data, columns=new_columns)

# Make sure the 'Purchased' column is present in the encoded_df DataFrame
if 'Purchased' in data.columns:

le = LabelEncoder()

if 'Purchased' in encoded_df.columns:
    encoded_df['Purchased'] = le.fit_transform(encoded_df['Purchased'])

# Q.) manhattan distance

import itertools

def manhattan_distance(point1, point2):
    return sum(abs(a - b) for a, b in zip(point1, point2))

def sum_manhattan_distances(points):
    total_sum = 0
    for pair in itertools.combinations(points, 2):
        total_sum += manhattan_distance(pair[0], pair[1])
    return total_sum

# Example usage
points = [
    (1, 1),
    (2, 3),
    (4, 2),
    (5, 5)
]

result = sum_manhattan_distances(points)
print(f"Sum of Manhattan distances between all pairs: {result}")

# Q.) weighted average specify axis (flattended array)

import numpy as np

def weighted_average(arr, weights, axis):

    # Reshape the weights to match the shape of the input array
    weights = np.reshape(weights, (-1, 1) if axis == 0 else (1, -1))

    # Compute the weighted sum
    weighted_sum = arr * weights

    # Compute the sum of the weights
    weight_sum = np.sum(weights, axis=axis)

    # Compute the weighted average
    avg = np.sum(weighted_sum, axis=axis) / weight_sum

    return avg

# Example usage:
arr = np.array([[1, 2, 3], [4, 5, 6]])
weights = np.array([0.1, 0.2, 0.3])
axis = 1

result = weighted_average(arr, weights, axis)
print(result)
